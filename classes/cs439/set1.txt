#include <stdlib.h>
#include <stdio.h>

void hello(int x){
	if ( x != 0 ) {
		printf("Hello, world!\n");
		hello(--x);
	} else {
		return;
	}
}

int main(){
	int x;
	printf("Enter a number: ");
	scanf("%d", &x);
	hello(x);

	if (x == x) {
		printf("Goodbye!\n");
	}
}
